ğŸŒ Production Deploymentâ€‹

# 1. Clone source code
git clone https://github.com/vietvo371/AgriTrace.git
cd AgriTrace

# 2. Backend Setup
cd backend
cp .env.example .env
# âœï¸ Fill in Database, Redis, JWT secret

# 3. Database Setup
mysql -u root -p
CREATE DATABASE agritrace;
CREATE USER 'agritrace_user'@'localhost' IDENTIFIED BY 'your_password';
GRANT ALL PRIVILEGES ON agritrace.* TO 'agritrace_user'@'localhost';
FLUSH PRIVILEGES;

# 4. Install Dependencies
yarn install

# 5. Run Migrations
yarn migrate

# 6. Seed Database
yarn seed

# 7. Start Production Server
yarn start:prod
